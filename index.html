// 核心修正部分：在 syncToCloud 中強化資料處理 async function syncToCloud() { if (isSyncing) return; isSyncing = true; updateStatus("🔄 同步中...", "info"); const currentData = buildDataFromUI(document.getElementById('tree-root')); // 確保 definedTags 是乾淨的陣列，避免傳送非預期對象 const cleanTags = Array.isArray(definedTags) ? definedTags : []; try { const { error, status, statusText } = await supabaseClient.from('bookmark_data').upsert({ id: 1, tree_json: currentData, defined_tags: cleanTags, // 確保格式正確 version: currentVersion, updated_at: new Date() }); if (error) { // 如果報 404 或找不到 Column，這通常是資料庫欄位沒開 console.error("Database Error:", error); updateStatus(`❌ 同步失敗: ${error.message}`, "danger"); return; } bookmarkTree = currentData; updateStatus("✅ 已同步", "success"); setTimeout(() => { if (!isSyncing) updateStatus("就緒", "info"); }, 2000); } catch (err) { updateStatus("❌ 網路或系統異常", "danger"); } finally { isSyncing = false; updateVersionDisplay(); } }
